```javascript
// tail
// 첫 번째 요소를 제외한 배열의 모든 요소를 반환합니다.
// 
// 배열의 길이가 1보다 큰 경우 Array.slice (1)을 반환하고, 그렇지 않으면 전체 배열을 반환합니다.

const tail = arr => (arr.length > 1 ? arr.slice(1) : arr);

// 예시
tail([1, 2, 3]); // [2,3]
tail([1]); // [1]
```
