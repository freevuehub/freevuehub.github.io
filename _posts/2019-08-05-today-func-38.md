```javascript
// splitLines
// 여러 줄 문자열을 줄 배열로 나눕니다.
// 
// 줄 바꿈을 일치시키고 배열을 만들려면 String.split ()과 정규 표현식을 사용하십시오.

const splitLines = str => str.split(/\r?\n/);

// 예시
splitLines('This\nis a\nmultiline\nstring.\n'); // ['This', 'is a', 'multiline', 'string.' , '']
```
