```javascript
// sum
// 두 개 이상의 숫자 / 배열의 합을 반환합니다.
// 
// Array.reduce()를 사용하여 각 값을 누적기에 추가하고 값 0으로 초기화합니다.

const sum = (...arr) => [...arr].reduce((acc, val) => acc + val, 0);

// 예시
sum(...[1, 2, 3, 4]); // 10
```
